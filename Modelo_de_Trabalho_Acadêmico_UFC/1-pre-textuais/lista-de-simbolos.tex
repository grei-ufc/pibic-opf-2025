\begin{simbolos} \itemsep -1pt
	\item[$ \mathcal{N} $] Conjunto de barras (nós) da rede
	\item[$ \mathcal{G} $] Conjunto de geradores
	\item[$ \mathcal{L} $] Conjunto de cargas
	\item[$ \mathcal{B} $] Conjunto de ramos AC (linhas e transformadores)
	\item[$ \mathcal{B}_{dc} $] Conjunto de elos HVDC
	\item[$ \mathcal{S} $] Conjunto de bancos \textit{shunt} chaveados
	\item[$ V_i $] Magnitude de tensão na barra $i$, em p.u.
	\item[$ \theta_i $] Ângulo de tensão na barra $i$, em radianos
	\item[$ p^g_k, q^g_k $] Potências ativa e reativa geradas pelo gerador $k$, em p.u.
	\item[$ p^d_l, q^d_l $] Potências ativa e reativa demandadas pela carga $l$, em p.u.
	\item[$ g_{ij}, b_{ij} $] Condutância e susceptância série do ramo $(i,j)$, em p.u.
	\item[$ b^{sh}_s $] Susceptância \textit{shunt} ajustável do banco $s$, em p.u.
	\item[$ t_{ij} $] Razão de transformação (módulo do \textit{tap}) do transformador no ramo $(i,j)$
	\item[$ s^v_i $] Variável de folga associada ao controle VLIM na barra $i$, em p.u.
	\item[$ s^q_l $] Variável de folga associada ao controle QLIM na carga $l$, em p.u.
	\item[$ \rho $] Penalidade associada às variáveis de folga na função objetivo
	\item[$ V^{\min}_i, V^{\max}_i $] Limites inferior e superior de magnitude de tensão na barra $i$
	\item[$ q^{g,\min}_k, q^{g,\max}_k $] Limites inferior e superior de geração reativa do gerador $k$
\end{simbolos}
